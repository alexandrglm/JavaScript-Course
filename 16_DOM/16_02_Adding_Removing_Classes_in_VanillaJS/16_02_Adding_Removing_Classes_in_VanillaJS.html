16-02 - Adding/Removing Classes in VanillaJS

Example 1: Adding Classes

<div id="demo1" class="demo-box">
    This is a demo box
</div>

// Add single class
const box1 = document.getElementById('demo1');
box1.classList.add('highlight');

// Add multiple classes
box1.classList.add('big-text', 'rounded');
// Console output: > box1.classList after additions: DOMTokenList(3) ["highlight", "big-text", "rounded"]
This is a demo box

Example 2: Removing Classes

<div id="demo2" class="demo-box highlight big-text rounded">
    This box has multiple classes
</div>

// Remove single class
const box2 = document.getElementById('demo2');
box2.classList.remove('highlight');

// Remove multiple classes
box2.classList.remove('big-text', 'rounded');
// Console output: > box2.classList after removals: DOMTokenList(0) []
This box has multiple classes

Example 3: Toggling Classes

<div id="demo3" class="demo-box">
    Click buttons to toggle my classes
</div>

const box3 = document.getElementById('demo3');

// Toggle class (add if missing, remove if present)
box3.classList.toggle('highlight');

// Force toggle (second parameter)
box3.classList.toggle('blue-bg', true);  // Always add
box3.classList.toggle('red-border', false); // Always remove
// Console output: > After first toggle (highlight): DOMTokenList(1) ["highlight"] > After forced toggles: DOMTokenList(2) ["highlight", "blue-bg"]
Click buttons to toggle my classes

Example 4: Checking Class Existence

<div id="demo4" class="demo-box highlight rounded">
    Check my classes
</div>

const box4 = document.getElementById('demo4');
const hasHighlight = box4.classList.contains('highlight');
const hasBigText = box4.classList.contains('big-text');

console.log('Has highlight?', hasHighlight);
console.log('Has big-text?', hasBigText);
// Console output: > Has highlight? true > Has big-text? false
Check my classes

Example 5: Practical Application

<button id="toggle-btn">Toggle Menu</button>
<div id="menu" class="hidden">Menu Content</div>

const toggleBtn = document.getElementById('toggle-btn');
const menu = document.getElementById('menu');

toggleBtn.addEventListener('click', () => {
    menu.classList.toggle('hidden');
    toggleBtn.textContent = menu.classList.contains('hidden') 
        ? 'Show Menu' : 'Hide Menu';
});
// Console output: > Clicking button toggles 'hidden' class > Button text updates accordingly
Menu Content
